226 
227  // Generate set of candidate positions for generation.
228  set<Pos> cands;
229  for (int i = 0; i < rows(); ++i) {
230  int j;
231  for (j = 0; j < cols() and cell(i, j).type == WATER; ++j) ;
232  if (j < cols()) cands.insert(Pos(i, j));
233 
234  for (j = cols()-1; j >= 0 and cell(i, j).type == WATER; --j) ;
235  if (j >= 0) cands.insert(Pos(i, j));
236  }
237 
238  // Regenerate killed units using valid candidate positions.
239  for (int id : gen) {
240  Pos pos = Pos(-1, -1);
241  while (pos == Pos(-1, -1) and not cands.empty()) {
242  int k = random(0, cands.size()-1);
243  auto it = cands.begin();
244  advance(it, k);
245  if (valid_to_spawn(*it)) pos = *it;
246  cands.erase(it);
247  }
248  if (pos == Pos(-1, -1)) // This should very very rarely happen.
249  for (int i = 0; i < rows() and pos == Pos(-1, -1); ++i)
250  for (int j = 0; j < cols() and pos == Pos(-1, -1); ++j)
251  if (valid_to_spawn(Pos(i, j)))
252  pos = Pos(i, j);
253  _my_assert(pos != Pos(-1, -1), "Cannot find a cell to regenerate units");
254  place(id, pos);
255  }
