248  template <class Prob>
249  vector<Pos> curve_from(int i0, int j0, Prob prob, bool allow_diags = true) {
250 
251  const vector<int>& diri = allow_diags ? DIRI8 : DIRI4;
252  const vector<int>& dirj = allow_diags ? DIRJ8 : DIRJ4;
253  const int D = diri.size();
254 
255  vector<Pos> curve;
256  vector<vector<bool>> mkd(rows(), vector<bool>(cols(), false));
257  int i = i0;
258  int j = j0;
259  int k = random(0, D-1);
260  while (true) {
261  curve.push_back({i, j});
262  mkd[i][j] = true;
263  int s, ii, jj, kk;
264  for (s = -2; s <= 1; ++s) {
265  if (s < -1) kk = k + random(-1, 1); // First try random
266  else kk = k + s; // then try exhaustively.
267  // -1 <= kk - k <= 1 to get a smooth shape.
268  kk = (D + kk) % D;
269  _my_assert(0 <= kk and kk < D, "In curve generation.");
270  ii = i + diri[kk];
271  jj = j + dirj[kk];
272  if (inside(ii, jj) and not mkd[ii][jj] and prob(i, j, ii, jj)) break;
273  }
274  if (s <= 1) { // Found a new point for the curve.
275  i = ii;
276  j = jj;
277  k = kk;
278  }
279  else break; // Could not continue the curve, so stop.
280  }
281  return curve;
282  }
